' ============================================================
' Decommissioning Cost Tool - Access Builder v0.6.2 All-in-One
' ============================================================
'
' v0.6.2 focus:
'   - Keeps the stable v0.5.2 no-tab-control design.
'   - Uses one main form with a category dropdown and a filtered datasheet subform.
'   - Keeps only the simple Word-compatible RTF export workflow.
'   - Replaces the three split reports with one combined report output:
'       rptJobEstimateReport
'   - Adds supporting subreports for the one combined report:
'       rptV062_CategoryBreakdown_Sub
'       rptV062_WbsSummary_Sub
'       rptV062_IncludedItems_Sub
'   - Adds visual Access relationships so the Relationships window is meaningful.
'   - Keeps relationships non-enforced to avoid breaking test databases during iteration.
'
' v0.6.2 fixes/additions:
'   - Fixes Word/RTF export Type mismatch by treating Access output format as Variant.
'   - Adds DefaultQuantity to tblPresetItems.
'   - Applying Radiological / Non-Radiological defaults now also fills Quantity.
'   - Default preset quantities are manually editable in tblPresetItems.
'   - Adds a Load Existing Job dropdown on the main form.
'
' How to use:
'   1. Open Access.
'   2. Create a blank .accdb, or use a copy of your current test .accdb.
'   3. Press Alt + F11.
'   4. Insert > Module.
'   5. Paste this whole module.
'   6. Run: BuildCostTool_V062_AllInOne
'
' Notes:
'   - Test on a copy first.
'   - No Python is needed at runtime.
'   - No tab control is used.
'   - Settings are still edited manually in tblSettings.
'   - Export creates one .rtf file that can be opened in Word.
'
' ============================================================
' ENTRY POINT
' ============================================================

Public Sub BuildCostTool_V062_AllInOne()
    On Error GoTo ErrHandler

    DoCmd.Hourglass True
    Application.Echo False

    V062_CreateCoreTables
    V062_UpdateSchemaForV062
    V062_CreateIndexes
    V062_CreateRelationships
    V062_SeedCategories
    V062_SeedSettings
    V062_SeedCostLibrary
    V062_SeedPresets
    V062_CreateSavedQueries
    V062_CreateStarterJobIfMissing "JOB-001", "New Decommissioning Estimate", ""

    V062_DeleteFormIfExists "frmJobEstimate"
    V062_DeleteFormIfExists "frmJobLinesSubform"

    ' v0.6 replaces the old three-report export with one combined report.
    V062_DeleteReportIfExists "rptJobEstimateReport"
    V062_DeleteReportIfExists "rptV062_CategoryBreakdown_Sub"
    V062_DeleteReportIfExists "rptV062_WbsSummary_Sub"
    V062_DeleteReportIfExists "rptV062_IncludedItems_Sub"

    ' Clean up older v0.6.1 supporting reports if this is an upgrade run.
    V062_DeleteReportIfExists "rptV061_CategoryBreakdown_Sub"
    V062_DeleteReportIfExists "rptV061_WbsSummary_Sub"
    V062_DeleteReportIfExists "rptV061_IncludedItems_Sub"

    ' Clean up older v0.5.2 reports if they exist.
    V062_DeleteReportIfExists "rptCostSummary"
    V062_DeleteReportIfExists "rptIncludedItems"
    V062_DeleteReportIfExists "rptWbsSummary"

    ' Clean up older v0.6 supporting reports if this is an upgrade run.
    V062_DeleteReportIfExists "rptV060_CategoryBreakdown_Sub"
    V062_DeleteReportIfExists "rptV060_WbsSummary_Sub"
    V062_DeleteReportIfExists "rptV060_IncludedItems_Sub"

    V062_CreateJobLinesSubform
    V062_CreateReports
    V062_CreateJobEstimateMainForm

    Application.Echo True
    DoCmd.Hourglass False

    DoCmd.OpenForm "frmJobEstimate"
    Forms("frmJobEstimate").Controls("cboCategory").value = "Equipment"
    V062_FilterJobLines

    MsgBox "Cost Tool v0.6.2 built successfully." & vbCrLf & vbCrLf & _
           "One combined Word/RTF report is available, visual relationships are added, and existing jobs can be loaded from the dropdown.", _
           vbInformation, "Build Complete"
    Exit Sub

ErrHandler:
    Application.Echo True
    DoCmd.Hourglass False
    MsgBox "Build failed: " & Err.Number & vbCrLf & Err.Description, vbCritical, "v0.6.2 Build Error"
End Sub

Private Sub V062_CreateTableIfMissing(ByVal tableName As String, ByVal ddl As String)
    If Not V062_TableExists(tableName) Then
        CurrentDb.Execute ddl, dbFailOnError
    End If
End Sub

Private Function V062_TableExists(ByVal tableName As String) As Boolean
    Dim tdf As DAO.TableDef
    For Each tdf In CurrentDb.TableDefs
        If StrComp(tdf.Name, tableName, vbTextCompare) = 0 Then
            V062_TableExists = True
            Exit Function
        End If
    Next tdf
    V062_TableExists = False
End Function


' ============================================================
' SCHEMA UPDATES FOR v0.6.2
' ============================================================

Private Sub V062_UpdateSchemaForV062()
    ' Existing v0.6 databases already have tblPresetItems, but without DefaultQuantity.
    ' This adds the column safely if it is missing, then gives old preset rows a simple default of 1.
    V062_AddFieldIfMissing "tblPresetItems", "DefaultQuantity", "DOUBLE"

    On Error Resume Next
    CurrentDb.Execute "UPDATE tblPresetItems SET DefaultQuantity=1 WHERE DefaultQuantity Is Null;", dbFailOnError
    On Error GoTo 0
End Sub

Private Sub V062_AddFieldIfMissing(ByVal tableName As String, ByVal fieldName As String, ByVal fieldTypeSql As String)
    If V062_FieldExists(tableName, fieldName) Then Exit Sub
    CurrentDb.Execute "ALTER TABLE " & tableName & " ADD COLUMN " & fieldName & " " & fieldTypeSql & ";", dbFailOnError
End Sub

Private Function V062_FieldExists(ByVal tableName As String, ByVal fieldName As String) As Boolean
    On Error GoTo MissingField
    Dim fld As DAO.Field
    Set fld = CurrentDb.TableDefs(tableName).Fields(fieldName)
    V062_FieldExists = True
    Exit Function
MissingField:
    V062_FieldExists = False
End Function

Private Sub V062_CreateIndexes()
    V062_CreateIndexIfMissing "tblCostLibrary", "idxCostLibrary_CategoryItem", _
        "CREATE INDEX idxCostLibrary_CategoryItem ON tblCostLibrary (CategoryName, ItemID);"

    V062_CreateIndexIfMissing "tblPresetItems", "idxPresetItems_Preset", _
        "CREATE INDEX idxPresetItems_Preset ON tblPresetItems (PresetName, CategoryName, ItemID);"

    V062_CreateIndexIfMissing "tblJobLines", "idxJobLines_Job", _
        "CREATE INDEX idxJobLines_Job ON tblJobLines (JobID, CategoryName, ItemID);"
End Sub

Private Sub V062_CreateIndexIfMissing(ByVal tableName As String, ByVal indexName As String, ByVal ddl As String)
    On Error GoTo CreateIt
    Dim idx As DAO.Index
    Set idx = CurrentDb.TableDefs(tableName).Indexes(indexName)
    Exit Sub
CreateIt:
    Err.Clear
    CurrentDb.Execute ddl, dbFailOnError
End Sub


' ============================================================
' VISUAL RELATIONSHIPS
' ============================================================
'
' These relationships are intentionally created as non-enforced visual
' relationships. That means they appear in Database Tools > Relationships,
' but they do not block testing if old scratch data is messy.
'
' This gives the database a proper visual map without adding risky cascade
' behaviour while the tool is still being iterated.

Private Sub V062_CreateRelationships()
    V062_CreateRelationshipIfMissing "relV060_Jobs_JobLines", _
        "tblJobs", "JobID", "tblJobLines", "JobID"

    V062_CreateRelationshipIfMissing "relV060_Categories_CostLibrary", _
        "tblCategories", "CategoryName", "tblCostLibrary", "CategoryName"

    V062_CreateRelationshipIfMissing "relV060_Categories_JobLines", _
        "tblCategories", "CategoryName", "tblJobLines", "CategoryName"

    V062_CreateRelationshipIfMissing "relV060_Categories_PresetItems", _
        "tblCategories", "CategoryName", "tblPresetItems", "CategoryName"

    V062_CreateRelationshipIfMissing "relV060_Presets_PresetItems", _
        "tblPresets", "PresetName", "tblPresetItems", "PresetName"

    V062_CreateRelationshipIfMissing "relV060_CostLibrary_JobLines", _
        "tblCostLibrary", "LibraryID", "tblJobLines", "LibraryID"
End Sub

Private Sub V062_CreateRelationshipIfMissing(ByVal relationName As String, ByVal parentTable As String, ByVal parentField As String, ByVal childTable As String, ByVal childField As String)
    On Error GoTo CleanFail

    Dim db As DAO.Database
    Dim rel As DAO.Relation
    Dim fld As DAO.Field

    If V062_RelationshipExists(relationName) Then Exit Sub

    Set db = CurrentDb
    Set rel = db.CreateRelation(relationName, parentTable, childTable, dbRelationDontEnforce)

    Set fld = rel.CreateField(parentField)
    fld.ForeignName = childField
    rel.Fields.Append fld

    db.Relations.Append rel
    db.Relations.Refresh
    Exit Sub

CleanFail:
    ' Relationship creation should not stop the whole builder.
    ' If a relationship cannot be added, the rest of the tool can still run.
    Err.Clear
End Sub

Private Function V062_RelationshipExists(ByVal relationName As String) As Boolean
    On Error GoTo NotFound
    Dim rel As DAO.Relation
    Set rel = CurrentDb.Relations(relationName)
    V062_RelationshipExists = True
    Exit Function
NotFound:
    V062_RelationshipExists = False
End Function

Private Sub V062_CreateSavedQueries()
    V062_SaveQuery "qryActiveCostLibrary", _
        "SELECT tblCostLibrary.* " & _
        "FROM tblCostLibrary INNER JOIN tblCategories ON tblCostLibrary.CategoryName = tblCategories.CategoryName " & _
        "WHERE tblCostLibrary.IsActive=True " & _
        "ORDER BY tblCategories.DisplayOrder, tblCostLibrary.ItemID, tblCostLibrary.ItemName;"

    V062_SaveQuery "qryJobLineAmounts", _
        "SELECT tblJobLines.JobLineID, tblJobLines.JobID, tblJobLines.LibraryID, tblJobLines.CategoryName, tblJobLines.ItemID, " & _
        "tblJobLines.IncludeItem, tblJobLines.WBSCode, tblJobLines.WBSSubCode, tblJobLines.ItemName, tblJobLines.Quantity, tblJobLines.UnitName, " & _
        "tblJobLines.BaseUnitRateUSD2009, " & _
        "V062_AdjustedRateAUD([BaseUnitRateUSD2009]) AS AdjustedRateAUD, " & _
        "V062_LineAmountAUD([IncludeItem],[Quantity],[BaseUnitRateUSD2009]) AS AmountAUD " & _
        "FROM tblJobLines INNER JOIN tblCategories ON tblJobLines.CategoryName = tblCategories.CategoryName " & _
        "ORDER BY tblJobLines.JobID, tblCategories.DisplayOrder, tblJobLines.ItemID, tblJobLines.ItemName;"

    V062_SaveQuery "qryJobLineEdit", _
        "SELECT tblJobLines.JobLineID, tblJobLines.JobID, tblJobLines.LibraryID, tblJobLines.CategoryName, tblJobLines.ItemID, " & _
        "tblJobLines.IncludeItem, tblJobLines.WBSCode, tblJobLines.WBSSubCode, tblJobLines.ItemName, tblJobLines.Quantity, tblJobLines.UnitName, " & _
        "tblJobLines.BaseUnitRateUSD2009, " & _
        "V062_AdjustedRateAUD([BaseUnitRateUSD2009]) AS AdjustedRateAUD, " & _
        "V062_LineAmountAUD([IncludeItem],[Quantity],[BaseUnitRateUSD2009]) AS AmountAUD " & _
        "FROM tblJobLines;"

    V062_SaveQuery "qryIncludedLineItems", _
        "SELECT q.* " & _
        "FROM qryJobLineAmounts AS q INNER JOIN tblCategories AS c ON q.CategoryName = c.CategoryName " & _
        "WHERE q.IncludeItem=True " & _
        "ORDER BY q.JobID, c.DisplayOrder, q.ItemID, q.ItemName;"

    V062_SaveQuery "qryCategoryTotals", _
        "SELECT q.JobID, q.CategoryName, Sum(q.AmountAUD) AS CategoryTotalAUD " & _
        "FROM qryJobLineAmounts AS q INNER JOIN tblCategories AS c ON q.CategoryName = c.CategoryName " & _
        "WHERE q.IncludeItem=True " & _
        "GROUP BY q.JobID, q.CategoryName, c.DisplayOrder " & _
        "ORDER BY q.JobID, c.DisplayOrder;"

    V062_SaveQuery "qryWbsSummary", _
        "SELECT JobID, WBSCode, WBSSubCode, Min(ItemName) AS ExampleItem, Sum(AmountAUD) AS WbsTotalAUD " & _
        "FROM qryJobLineAmounts " & _
        "WHERE IncludeItem=True " & _
        "GROUP BY JobID, WBSCode, WBSSubCode " & _
        "ORDER BY JobID, WBSCode, WBSSubCode;"

    V062_SaveQuery "qryGrandTotals", _
        "SELECT JobID, Sum(AmountAUD) AS SubtotalAUD, " & _
        "Sum(AmountAUD) * V062_GetSettingDbl('ContingencyPct',0) / 100 AS ContingencyAUD, " & _
        "Sum(AmountAUD) * (1 + V062_GetSettingDbl('ContingencyPct',0) / 100) AS GrandTotalAUD " & _
        "FROM qryJobLineAmounts " & _
        "WHERE IncludeItem=True " & _
        "GROUP BY JobID;"

    V062_SaveQuery "qryReportCostSummary", _
        "SELECT tblJobs.JobID, tblJobs.JobName, tblJobs.PreparedBy, Now() AS GeneratedAt, " & _
        "V062_GetSettingDbl('InflationFactor',1) AS InflationFactor, " & _
        "V062_GetSettingDbl('UsdToAudRate',1) AS UsdToAudRate, " & _
        "V062_GetSettingDbl('OverheadPct',0) AS OverheadPct, " & _
        "V062_GetSettingDbl('ContingencyPct',0) AS ContingencyPct, " & _
        "Nz(qryGrandTotals.SubtotalAUD,0) AS SubtotalAUD, " & _
        "Nz(qryGrandTotals.ContingencyAUD,0) AS ContingencyAUD, " & _
        "Nz(qryGrandTotals.GrandTotalAUD,0) AS GrandTotalAUD " & _
        "FROM tblJobs LEFT JOIN qryGrandTotals ON tblJobs.JobID = qryGrandTotals.JobID;"

    ' Optional helper query for manually setting preset quantities.
    ' Open this query and edit DefaultQuantity, then apply the preset to a job.
    V062_SaveQuery "qryPresetQuantityEditor", _
        "SELECT pi.PresetName, pi.CategoryName, pi.ItemID, cl.ItemName, cl.UnitName, pi.DefaultQuantity " & _
        "FROM tblPresetItems AS pi LEFT JOIN tblCostLibrary AS cl " & _
        "ON (pi.CategoryName = cl.CategoryName) AND (pi.ItemID = cl.ItemID) " & _
        "ORDER BY pi.PresetName, pi.CategoryName, pi.ItemID;"
End Sub

Public Function V062_GetSettingDbl(ByVal settingName As String, Optional ByVal defaultValue As Double = 0) As Double
    On Error GoTo Fallback
    Dim v As Variant
    v = DLookup("SettingValueNumber", "tblSettings", "SettingName=" & V062_Q(settingName))
    If IsNull(v) Or v = "" Then
        V062_GetSettingDbl = defaultValue
    Else
        V062_GetSettingDbl = CDbl(v)
    End If
    Exit Function
Fallback:
    V062_GetSettingDbl = defaultValue
End Function

Public Function V062_AdjustedRateAUD(ByVal baseUnitRateUSD2009 As Variant) As Currency
    Dim baseRate As Double
    baseRate = Nz(baseUnitRateUSD2009, 0)

    V062_AdjustedRateAUD = CCur(baseRate * _
        V062_GetSettingDbl("InflationFactor", 1) * _
        V062_GetSettingDbl("UsdToAudRate", 1) * _
        (1 + V062_GetSettingDbl("OverheadPct", 0) / 100))
End Function

Public Function V062_LineAmountAUD(ByVal includeItem As Variant, ByVal quantity As Variant, ByVal baseUnitRateUSD2009 As Variant) As Currency
    If Nz(includeItem, False) = False Then
        V062_LineAmountAUD = 0
    Else
        V062_LineAmountAUD = CCur(Nz(quantity, 0) * V062_AdjustedRateAUD(baseUnitRateUSD2009))
    End If
End Function

Public Sub V062_CreateStarterJobIfMissing(ByVal jobID As String, ByVal jobName As String, ByVal preparedBy As String)
    If DCount("*", "tblJobs", "JobID=" & V062_Q(jobID)) > 0 Then Exit Sub
    V062_CreateNewJobFromLibrary jobID, jobName, preparedBy
End Sub

Public Sub V062_CreateNewJobFromLibrary(ByVal jobID As String, ByVal jobName As String, ByVal preparedBy As String)
    If DCount("*", "tblJobs", "JobID=" & V062_Q(jobID)) > 0 Then
        Err.Raise vbObjectError + 1001, , "JobID already exists: " & jobID
    End If

    CurrentDb.Execute "INSERT INTO tblJobs (JobID, JobName, PreparedBy, CreatedAt, UpdatedAt) VALUES (" & _
        V062_Q(jobID) & ", " & V062_Q(jobName) & ", " & V062_Q(preparedBy) & ", Now(), Now());", dbFailOnError

    CurrentDb.Execute "INSERT INTO tblJobLines " & _
        "(JobID, LibraryID, CategoryName, ItemID, IncludeItem, WBSCode, WBSSubCode, ItemName, Quantity, UnitName, BaseUnitRateUSD2009) " & _
        "SELECT " & V062_Q(jobID) & " AS JobID, LibraryID, CategoryName, ItemID, False AS IncludeItem, WBSCode, WBSSubCode, ItemName, 0 AS Quantity, UnitName, BaseUnitRateUSD2009 " & _
        "FROM qryActiveCostLibrary;", dbFailOnError
End Sub

Public Sub V062_ApplyPresetToJob(ByVal jobID As String, ByVal presetName As String)
    If DCount("*", "tblJobs", "JobID=" & V062_Q(jobID)) = 0 Then
        Err.Raise vbObjectError + 1002, , "JobID not found: " & jobID
    End If

    ' Reset the job first so changing preset does not leave stale quantities behind.
    CurrentDb.Execute "UPDATE tblJobLines SET IncludeItem=False, Quantity=0 WHERE JobID=" & V062_Q(jobID) & ";", dbFailOnError

    ' Include preset items and copy their editable DefaultQuantity into the job Quantity.
    CurrentDb.Execute "UPDATE tblJobLines INNER JOIN tblPresetItems " & _
        "ON (tblJobLines.CategoryName = tblPresetItems.CategoryName) AND (tblJobLines.ItemID = tblPresetItems.ItemID) " & _
        "SET tblJobLines.IncludeItem=True, tblJobLines.Quantity=Nz(tblPresetItems.DefaultQuantity,1) " & _
        "WHERE tblJobLines.JobID=" & V062_Q(jobID) & " AND tblPresetItems.PresetName=" & V062_Q(presetName) & ";", dbFailOnError
End Sub

Public Sub V062_ClearJobInputs(ByVal jobID As String)
    CurrentDb.Execute "UPDATE tblJobLines SET IncludeItem=False, Quantity=0 WHERE JobID=" & V062_Q(jobID) & ";", dbFailOnError
End Sub

Public Function V062_FilterJobLines() As Boolean
    On Error GoTo ErrHandler

    Dim jobID As String
    Dim categoryName As String
    Dim filterText As String

    jobID = V062_CurrentJobID()
    categoryName = Nz(Forms("frmJobEstimate").Controls("cboCategory").value, "")

    If categoryName = "" Then
        categoryName = "Equipment"
        Forms("frmJobEstimate").Controls("cboCategory").value = categoryName
    End If

    If jobID = "" Then Exit Function

    On Error Resume Next
    Forms("frmJobEstimate").Controls("cboJobSelector").value = jobID
    On Error GoTo ErrHandler

    filterText = "JobID=" & V062_Q(jobID) & " AND CategoryName=" & V062_Q(categoryName)

    With Forms("frmJobEstimate").Controls("subJobLines").Form
        .Filter = filterText
        .FilterOn = True
        .Requery
    End With

    Forms("frmJobEstimate").Recalc
    V062_FilterJobLines = True
    Exit Function

ErrHandler:
    V062_FilterJobLines = False
End Function

Public Function V062_UI_Refresh() As Boolean
    On Error Resume Next
    V062_FilterJobLines
    V062_RefreshJobSelector
    Forms("frmJobEstimate").Recalc
    Forms("frmJobEstimate").Repaint
    V062_UI_Refresh = True
End Function

Public Function V062_UI_LoadSelectedJob() As Boolean
    On Error GoTo ErrHandler

    Dim selectedJobID As String
    Dim rs As DAO.Recordset

    selectedJobID = Nz(Forms("frmJobEstimate").Controls("cboJobSelector").value, "")
    If selectedJobID = "" Then Exit Function

    Set rs = Forms("frmJobEstimate").RecordsetClone
    rs.FindFirst "JobID=" & V062_Q(selectedJobID)

    If Not rs.NoMatch Then
        Forms("frmJobEstimate").Bookmark = rs.Bookmark
    End If

    rs.Close

    If Nz(Forms("frmJobEstimate").Controls("cboCategory").value, "") = "" Then
        Forms("frmJobEstimate").Controls("cboCategory").value = "Equipment"
    End If

    V062_UI_Refresh

    V062_UI_LoadSelectedJob = True
    Exit Function

ErrHandler:
    MsgBox "Could not load selected job:" & vbCrLf & Err.Description, vbExclamation, "Load Job Error"
    V062_UI_LoadSelectedJob = False
End Function

Private Sub V062_RefreshJobSelector()
    On Error Resume Next

    With Forms("frmJobEstimate").Controls("cboJobSelector")
        .RowSource = "SELECT JobID, JobName FROM tblJobs ORDER BY UpdatedAt DESC, CreatedAt DESC, JobID;"
        .Requery
        .value = V062_CurrentJobID()
    End With
End Sub

Public Function V062_UI_NewJob() As Boolean
    On Error GoTo ErrHandler

    Dim jobID As String
    Dim jobName As String
    Dim preparedBy As String
    Dim rs As DAO.Recordset

    jobID = Trim(InputBox("Enter a new Job ID.", "New Job", "JOB-" & Format(Now(), "yyyymmdd-hhnn")))
    If jobID = "" Then Exit Function

    jobName = Trim(InputBox("Enter the job name.", "New Job", "New Decommissioning Estimate"))
    If jobName = "" Then jobName = "New Decommissioning Estimate"

    preparedBy = Trim(InputBox("Prepared by:", "New Job", ""))

    V062_CreateNewJobFromLibrary jobID, jobName, preparedBy

    Forms("frmJobEstimate").Requery

    Set rs = Forms("frmJobEstimate").RecordsetClone
    rs.FindFirst "JobID=" & V062_Q(jobID)
    If Not rs.NoMatch Then
        Forms("frmJobEstimate").Bookmark = rs.Bookmark
    End If
    rs.Close

    V062_RefreshJobSelector
    Forms("frmJobEstimate").Controls("cboCategory").value = "Equipment"
    V062_UI_Refresh

    MsgBox "Created new job: " & jobID, vbInformation, "New Job"
    V062_UI_NewJob = True
    Exit Function

ErrHandler:
    MsgBox "Could not create new job:" & vbCrLf & Err.Description, vbExclamation, "New Job Error"
    V062_UI_NewJob = False
End Function

Public Function V062_UI_ExportRTF() As Boolean
    On Error GoTo ErrHandler

    Dim jobID As String
    Dim outputFolder As String
    Dim outputPath As String

    jobID = V062_CurrentJobID()
    If jobID = "" Then Exit Function

    V062_UI_Refresh

    outputFolder = CurrentProject.Path
    outputPath = outputFolder & "\" & V062_SafeFilePart(jobID) & "_Decommissioning_Cost_Report.rtf"

    V062_OutputFilteredReport "rptJobEstimateReport", "JobID=" & V062_Q(jobID), acFormatRTF, outputPath

    MsgBox "Word-compatible report exported to:" & vbCrLf & outputPath & vbCrLf & vbCrLf & _
           "Open the .rtf file in Word, then save as .docx or PDF if needed.", _
           vbInformation, "Export Word Report Complete"

    V062_UI_ExportRTF = True
    Exit Function

ErrHandler:
    MsgBox "Word report export failed:" & vbCrLf & Err.Description, vbExclamation, "Export Word Report Error"
    V062_UI_ExportRTF = False
End Function

Private Sub V062_OutputFilteredReport(ByVal reportName As String, ByVal whereText As String, ByVal outputFormat As Variant, ByVal outputPath As String)
    On Error GoTo CleanFail

    DoCmd.OpenReport reportName, acViewPreview, , whereText, acHidden
    DoCmd.OutputTo acOutputReport, reportName, outputFormat, outputPath, False
    DoCmd.Close acReport, reportName, acSaveNo
    Exit Sub

CleanFail:
    On Error Resume Next
    DoCmd.Close acReport, reportName, acSaveNo
    Err.Raise Err.Number, , Err.Description
End Sub

Private Function V062_SafeFilePart(ByVal textValue As String) As String
    Dim s As String
    s = textValue

    s = Replace(s, "\", "_")
    s = Replace(s, "/", "_")
    s = Replace(s, ":", "_")
    s = Replace(s, "*", "_")
    s = Replace(s, "?", "_")
    s = Replace(s, """", "_")
    s = Replace(s, "<", "_")
    s = Replace(s, ">", "_")
    s = Replace(s, "|", "_")
    s = Replace(s, " ", "_")

    V062_SafeFilePart = s
End Function

Private Function V062_Q(ByVal value As Variant) As String
    If IsNull(value) Then
        V062_Q = "Null"
    Else
        V062_Q = "'" & Replace(CStr(value), "'", "''") & "'"
    End If
End Function

Private Function V062_SqlNum(ByVal value As Double) As String
    V062_SqlNum = Replace(CStr(value), ",", ".")
End Function
